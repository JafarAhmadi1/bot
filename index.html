'; Object.keys(FRAME_IMAGE_URLS).forEach((key, index) => { const button = document.createElement('button'); button.innerText = `🖼️ قاب ${index + 1}`; button.className = 'action-button'; button.onclick = () => { frameSelectionDiv.querySelectorAll('button').forEach(btn => btn.disabled = true); statusDiv.innerText = "⏳ در حال ساخت تصویر شما... لطفاً صبر کنید."; createFramedImage(userImageObjectUrl, FRAME_IMAGE_URLS[key]); }; frameSelectionDiv.appendChild(button); }); frameSelectionDiv.style.display = 'block'; } async function createFramedImage(userImageUrl, frameUrl) { const ctx = canvas.getContext('2d'); const loadImage = (src) => new Promise((resolve, reject) => { const img = new Image(); img.crossOrigin = "Anonymous"; img.onload = () => resolve(img); img.onerror = reject; img.src = src; }); try { const [userImg, frameImg] = await Promise.all([loadImage(userImageUrl), loadImage(frameUrl)]); const size = 512; canvas.width = size; canvas.height = size; const imgAspectRatio = userImg.naturalWidth / userImg.naturalHeight; const canvasAspectRatio = 1; let drawWidth, drawHeight, x, y; if (imgAspectRatio > canvasAspectRatio) { drawHeight = size; drawWidth = drawHeight * imgAspectRatio; x = (size - drawWidth) / 2; y = 0; } else { drawWidth = size; drawHeight = drawWidth / imgAspectRatio; x = 0; y = (size - drawHeight) / 2; } ctx.drawImage(userImg, x, y, drawWidth, drawHeight); ctx.drawImage(frameImg, 0, 0, size, size); const finalImageDataUrl = canvas.toDataURL('image/png'); resultImage.src = finalImageDataUrl; frameSelectionDiv.style.display = 'none'; statusDiv.style.display = 'none'; resultContainer.style.display = 'block'; if (isEitaaEnv) { EitaaApp.MainButton.show(); } } catch (error) { statusDiv.innerText = "خطا در بارگذاری تصاویر. لطفاً دوباره تلاش کنید."; console.error("Image loading error:", error); if(isEitaaEnv) { EitaaApp.showAlert("خطا در بارگذاری تصاویر. لطفاً از اتصال اینترنت خود مطمئن شوید و دوباره تلاش کنید."); } resetApp(); } } // --- تابع جدید برای ذخیره تصویر --- function handleSaveImage() { // ابتدا یک پیام راهنما به کاربر نمایش می‌دهیم if (isEitaaEnv) { EitaaApp.showAlert("برای ذخیره، در صفحه‌ای که باز می‌شود، تصویر را چند لحظه لمس کرده و گزینه 'ذخیره تصویر' یا 'Download Image' را انتخاب کنید."); // دریافت تصویر از canvas به صورت data URL const imageDataUrl = canvas.toDataURL('image/png'); // باز کردن لینک تصویر در مرورگر خارجی EitaaApp.openLink(imageDataUrl); } else { // رفتار عادی برای مرورگر وب const link = document.createElement('a'); link.download = 'parcham-bala-profile.png'; link.href = canvas.toDataURL('image/png'); link.click(); } } // --- پایان تابع جدید --- function resetApp() { uploadButton.style.display = 'inline-block'; frameSelectionDiv.style.display = 'none'; resultContainer.style.display = 'none'; statusDiv.style.display = 'block'; statusDiv.innerText = ''; uploadInput.value = ''; if (userImageObjectUrl) { URL.revokeObjectURL(userImageObjectUrl); userImageObjectUrl = null; } if (isEitaaEnv) { EitaaApp.MainButton.hide(); } } });
